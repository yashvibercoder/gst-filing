### The Bug Fix Layer: `portal/backend/app/services/bug_fixes.py`

```python
"""
bug_fixes.py вҖ” Post-processing fixes applied AFTER generators run.
These fix known issues from the Delhi Audit without modifying System A.

Fixes applied:
  1. B2B/CDNR deduplication (keep E-Invoice version, drop Amazon duplicate)
  2. Place Of Supply normalization (strip spaces: "XX - Name" вҶ’ "XX-Name")
  3. HSN B2C floating point precision (round to 2 decimal places)
  4. UQC long code mapping (PCS вҶ’ PCS-PIECES, OTH вҶ’ OTH-OTHERS)
  5. Documents completion (derive series from B2B + CDNR data)
"""

class BugFixer:
    def fix_all(self, state_dir: str) -> FixReport:
        """Apply all fixes to CSVs in a state directory."""
        report = FixReport()
        report += self.fix_b2b_dedup(state_dir)
        report += self.fix_cdnr_dedup(state_dir)
        report += self.fix_pos_format(state_dir)
        report += self.fix_hsn_precision(state_dir)
        report += self.fix_uqc_codes(state_dir)
        report += self.fix_docs_completion(state_dir)
        return report

    def fix_b2b_dedup(self, state_dir: str) -> FixResult:
        """
        Deduplicate B2B invoices.
        Same Invoice Number from Amazon + E-Invoice = keep E-Invoice
        (E-Invoice has Receiver Name, Amazon doesn't)
        """

    def fix_pos_format(self, state_dir: str) -> FixResult:
        """
        Normalize Place Of Supply:
        "23 - Madhya Pradesh" вҶ’ "23-Madhya Pradesh"
        Applied to: b2b,sez,de.csv, cdnr1.csv
        """

    def fix_hsn_precision(self, state_dir: str) -> FixResult:
        """
        Round all numeric columns to 2 decimal places.
        Applied to: hsn(b2c).csv
        Fixes: 31445.659999999993 вҶ’ 31445.66
        """

    def fix_uqc_codes(self, state_dir: str) -> FixResult:
        """
        Map short UQC codes to GSTN long format:
        PCS вҶ’ PCS-PIECES, OTH вҶ’ OTH-OTHERS, UNT вҶ’ UNT-UNITS
        KGS вҶ’ KGS-KILOGRAMS, MTR вҶ’ MTR-METERS, LTR вҶ’ LTR-LITRES
        Applied to: hsn(b2b).csv, hsn(b2c).csv
        """

    def fix_docs_completion(self, state_dir: str) -> FixResult:
        """
        Derive document series from B2B + CDNR data.
        Extract min/max invoice numbers per prefix (FABBQS, IN-, KNYT-)
        Count per series.
        Applied to: docs.csv
        """
```
